Metaprogramming

example: Memoize

  • The following code:
    • 
using Memoize
@memoize function ex2(n)
    println("Evaluating...")
    return sum(sqrt, 1:n)
end
  • expands to (at compilation time): 
    
function ex2_unmemoized(n)
    println("Evaluating...")
    return sum(sqrt, 1:n)
end

empty!(ObjectIdDict())

ex2(n)::Any = begin
    if haskey(ObjectIdDict(), (n,))
        (ObjectIdDict())[(n,)]::
          Core.Inference.return_type(
            ex2_unmemoized, typeof((n,)))
    else 
        (ObjectIdDict())[(n,)] = ex2_unmemoized(n)
    end
end
