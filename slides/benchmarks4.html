function ex2(n)
    # Note the `.` after `sqrt`
    return sum(sqrt.(1:n))
end

dummy = ex2(5)  # warm-up
@time println(ex2(500_000_000))
Improved memory usage: 

#-----------------------------------------
function ex3(n)
    return sum(sqrt, 1:n)
end

dummy = ex2(5)  # warm-up
@time println(ex2(500_000_000))
