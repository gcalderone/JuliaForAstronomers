import numpy as np
import math
import time

def ex2(n):
    a = np.arange(n)+1
    a = np.sqrt(a)
    return np.sum(a)

elapsed = time.time()
print(ex2(500000000))
elapsed = time.time() - elapsed
print("Elapsed time: ", elapsed)
