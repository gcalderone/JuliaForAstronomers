C: 

//----------------------------------------
#include <stdio.h>
#include <math.h>

double ex1(long int n) {
    long int i;
    double sum = 0.;
    for (i=1; i<=n; i++)
        sum += sqrt(i);
    return sum;
}

int main() {
    printf("%f\n", ex1(500000000));
}
Julia: 

#-----------------------------------------
function ex1(n)
    sum = 0.
    for i in 1:n
        sum += sqrt(i)
    end
    return sum
end

dummy = ex1(2)
@time println(ex1(500_000_000))
