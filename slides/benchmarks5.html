function ex2(n)
    return sum(sqrt, 1:n)
end

dummy = ex2(2)
@time println(ex2(500_000_000))
Julia: 

#-----------------------------------------
using Base.Threads
function ex4(n)
    arr = fill(0., nthreads())
   @threads for i = 1:n   
        arr[threadid()] += sqrt(float(i))
    end
    return sum(arr)
end  

dummy = ex4(10)
@time println(ex4(500_000_000))
