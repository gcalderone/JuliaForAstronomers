import math
import time

def ex1(n):
    sum = 0.
    for i in range(1,n):
        sum += math.sqrt(i)
    return sum

elapsed = time.time()
print(ex1(500000000))
elapsed = time.time() - elapsed
print("Elapsed time: ", elapsed)
